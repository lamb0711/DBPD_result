+        class PathMapElementCounter implements PathMap.ElementVisitor {
+            int count;
+            public void elementVisited(PathMap.Element element) {
+                LRUEntry mappedEntry = (LRUEntry) element.get();

+                if (cachedEntry == null) {
+                    String msg = "Path element (" + element +
