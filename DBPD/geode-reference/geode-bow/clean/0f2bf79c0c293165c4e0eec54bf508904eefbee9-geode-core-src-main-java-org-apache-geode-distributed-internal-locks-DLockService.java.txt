GEODE-6380: Annotating static fields

Marking static fields as immutable or with an annotation that indicates
how the field should be fixed/removed.

+import org.apache.geode.annotations.Immutable;
+import org.apache.geode.annotations.internal.MakeNotStatic;
+  @MakeNotStatic("This object is actually immutable ... but it uses singletons")
+  @Immutable
+  @Immutable
+  @MakeNotStatic
+  @MakeNotStatic
+  @MakeNotStatic
+  @MakeNotStatic
+  @Immutable
