GEODE-4073: Add ability for SuiteRunner to interleave a candidate test class (#1150)

- By specifying a class with the annotation SuiteRunner.Candidate you can
  interleave the candidate with each class defined in SuiteClasses.
+import java.lang.annotation.ElementType;
+import java.lang.annotation.Inherited;
+import java.lang.annotation.Retention;
+import java.lang.annotation.RetentionPolicy;
+import java.lang.annotation.Target;
+  @Retention(RetentionPolicy.RUNTIME)
+  @Target(ElementType.TYPE)
+  @Inherited
+  public @interface Candidate {
+    /**
+     * @return the classes to be run
+     */
+    public Class<?> value();
+  }
+
+
+    SuiteRunner.Candidate candidate = testClass.getAnnotation(Candidate.class);
+    Class<?> candidateClass = null;
+    if (candidate != null) {
+      candidateClass = candidate.value();
+    }
+
+
+      if (candidateClass != null) {
+        runners.add(new SuiteBlockRunner(testClass, candidateClass));
+      }
