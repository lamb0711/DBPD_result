+  @Override
+  public boolean equals(Object that) {
+    if (this == that) {
+      return true;
+    }
+    if (that == null || getClass() != that.getClass()) {
+      return false;
+    }
+    RegionConfig config = (RegionConfig) that;
+    return Objects.equal(getName(), config.getName()) &&
