+    public PluginManagement getPluginManagement()
+    {
+        PluginManagement pluginMgmt = null;
+

+        {

+        }
+

+    public void addPlugin( Plugin plugin )
+    {
+        Build build = model.getBuild();
+
+        if ( build == null )
+        {
+            build = new Build();
+
+            model.setBuild( build );
+        }
+
+        build.addPlugin( plugin );
+    }
