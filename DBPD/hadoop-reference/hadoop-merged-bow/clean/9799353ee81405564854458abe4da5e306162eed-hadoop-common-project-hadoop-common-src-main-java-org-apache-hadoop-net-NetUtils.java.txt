+    // Check if the address is defined on any interface

+      try {
+        local = NetworkInterface.getByInetAddress(addr) != null;
+      } catch (SocketException e) {
+        local = false;
+      }
+    }
+        System.out.println("address " + addr + " is local " + local);
+    return local;
+  }
