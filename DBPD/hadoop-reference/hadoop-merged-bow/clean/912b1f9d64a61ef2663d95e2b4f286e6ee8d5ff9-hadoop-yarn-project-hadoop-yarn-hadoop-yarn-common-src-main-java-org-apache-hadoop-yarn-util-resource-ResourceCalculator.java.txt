+  /**
+   * Divides lhs by rhs.
+   * If both lhs and rhs are having a value of 0, then we return 0.

+   * as the result.
+   * @param lhs
+   * @param rhs
+   * @return

+  public static float divideSafelyAsFloat(long lhs, long rhs) {
+    if (lhs == 0 && rhs == 0) {

+    } else {
+      return (float) lhs / (float) rhs;
+    }
+  }
