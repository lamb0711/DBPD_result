+  static List<EditLogFile> matchEditLogs(File[] filesInStorage) {

+      String name = f.getName();
+      // Check for edits

+      if (editsMatch.matches()) {
+        try {
+          long startTxId = Long.valueOf(editsMatch.group(1));
