+import org.apache.hadoop.conf.Configuration;

+      "container.vcores";
+  public static final int AM_CONTAINER_VCORES_DEFAULT = 1;

+        conf.getInt(AM_CONTAINER_VCORES, AM_CONTAINER_VCORES_DEFAULT));
