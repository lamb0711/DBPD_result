+  public static String camelize(String s) {
+    StringBuilder sb = new StringBuilder();
+    String[] words = split(s.toLowerCase(Locale.US), ESCAPE_CHAR, '_');
+

+      sb.append(org.apache.commons.lang.StringUtils.capitalize(word));
